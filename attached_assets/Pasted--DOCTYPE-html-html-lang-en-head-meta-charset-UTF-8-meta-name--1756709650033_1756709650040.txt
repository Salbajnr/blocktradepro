            // Back button

            // Show different transaction statuses
            function showStatus(status) {
                // Hide all status containers
                document.querySelectorAll('.transaction-status').forEach(el => {
                    el.classList.add('hidden');
                });

                // Show selected status
                document.getElementById(`${status}-transaction`).classList.remove('hidden');
            }

            // Copy to clipboard
            function copyToClipboard(text) {
                navigator.clipboard.writeText(text).then(() => {
                    // Show a temporary notification
                    const button = event.target.closest('button');
                    const originalIcon = button.innerHTML;
                    button.innerHTML = '<i class="fas fa-check text-green-500"></i>';

                    setTimeout(() => {
                        button.innerHTML = originalIcon;
                    }, 2000);
                }).catch(err => {
                    console.error('Could not copy text: ', err);
                });
            }

            // Print receipt
            function printReceipt() {
                window.print();
            }

            // Share transaction
            function shareTransaction() {
                if (navigator.share) {
                    navigator.share({
                        title: 'BlockTrade Transaction',
                        text: 'Transaction completed successfully',
                        url: window.location.href
                    });
                } else {
                    // Fallback for browsers that don't support Web Share API
                    copyToClipboard(window.location.href);
                    alert('Transaction link copied to clipboard');
                }
            }

                    alert('Transaction cancellation would be processed here');

            }

            // Refresh status
            function refreshStatus() {
                // Show loading state
                const button = event.target;
                const originalText = button.innerHTML;
                button.innerHTML = '<i class="fas fa-spinner fa-spin mr-2"></i>Refreshing...';
                button.disabled = true;

                // Simulate API call
                setTimeout(() => {
                    button.innerHTML = originalText;
                    button.disabled = false;
                    alert('Status refreshed');
                }, 2000);
            }

                if (confirm('Would you like to retry this transaction?')) {
                    window.location.href = '/deposit';

                }
